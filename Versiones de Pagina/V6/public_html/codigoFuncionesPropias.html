        	function miConcat(array1, array2){
        		
        		var longArray1=0;
        		var longArray2=0;
        		
        		for (var i in array1){
        			longArray1++;
        		}
        		
        		for (var i in array2){
        			longArray2++;
        		}
        		
        		/*alert(longArray1);
        		alert(longArray2);*/
        		var longAux = longArray1+longArray2;
        		
        		var array3= new Array(longAux);
        		
    			for (var i = 0; i< longArray1;i++){
    				array3[i] = array1[i];
    			}
        		
        		var iAux = 0;
        		for (var index = longArray1 ; index<longAux;index++ ){
        			array3[index]=array2[iAux];
        			iAux++;
        		}
        		return array3;
        	}

	     function miSplitC(){
       		
       		var cadena=document.getElementById("cadenasplit");
       		var separador=document.getElementById("separadorSplit");
       		var res=document.getElementById("resSplit");
       		
       		
       		cadena=cadena.value;
       		separador=separador.value;
       		
        	var cont=0;
        	var arr;
        	var arr2="";
        	var posicion=0;
        	var longSeparador=0;
        	var longCadena=0;
        	
        	for(var i in separador){
        		longSeparador++;
        	}
        	
        	for(var i in cadena){	
        		for(var a=0;a<longSeparador;a++){
               			arr2+=cadena[i++];
        		        		}
        		if(arr2==separador){
        			cont++;
        			arr2="";
        		}
        		else{
        			arr2="";
        		}
        		longCadena++;
        	}
        	
        	
        	arr=new Array(cont+1);
        	
        	arr[posicion]="";
        	for(var i=0;i<longCadena; i++){
        		
        		for(var a=0;a<longSeparador;a++){
        			arr2+=cadena[i++];
        			
        		}
        		
        		if(arr2!=separador){
        			i-=longSeparador; 
        			
        			arr[posicion]+=cadena[i];
        			arr2="";
        		}
        		
        		else{
        				
					 
        			if((i-longSeparador)==0  || i==longCadena){
        				arr2="";
        				i-=1;
	                   
        			}
        			else{
        			
        			posicion++;
        			arr[posicion]="";
        			arr2="";
        			i-=1;
        			}
        		}
        	}
    
        	for(i in arr){
        	res.value+="posicion "+i+ "= "+arr[i]+"\\n";
        }
        
        }
